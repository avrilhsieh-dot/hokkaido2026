                { dayNum: 7, dateShort: '7/9', title: '札幌啤酒與血拼', area: '札幌', events: [{ time: '14:35', title: '札幌啤酒博物館', icon: 'fas fa-beer', lat: 43.07, lng: 141.36, guide: '了解歷史' }] },

                { dayNum: 8, dateShort: '7/10', title: '熱血觀賽之旅', area: '札幌', events: [{ time: '18:00', title: 'Es Con Field 賽事', icon: 'fas fa-baseball', lat: 42.99, lng: 141.54, guide: '火腿隊賽事' }] },

                { dayNum: 9, dateShort: '7/11', title: '最後採買與歸途', area: '新千歲', events: [{ time: '11:10', title: '新千歲機場', icon: 'fas fa-plane', lat: 42.77, lng: 141.68, guide: '回程' }] }

            ]);


            const currentDayData = computed(() => itinerary.value[currentDay.value]);

            const expenses = ref([]);

            const showExpenseModal = ref(false);

            const expenseForm = ref({ shop: '', category: '飲食', amountJPY: '', amountTWD: '' });

            const luggagePeople = ['家禎', '宇淳', '侑侑', '懷懷', '媽媽'];

            const luggageCategories = ['行李&證件', '衣/配件', '日常、盥洗', '3C/其他'];

            const currentLuggagePerson = ref('家禎');

            const luggageList = ref([]);

            const showLuggageModal = ref(false);

            const luggageForm = ref({ person: '家禎', category: '行李&證件', name: '', quantity: 1, checked: false });


            const selectDay = (index) => {

                currentDay.value = index;

                if (currentTab.value === 'map') updateMapMarkers();

            };


            const initMap = () => {

                if (map) return;

                map = window.L.map('map', { zoomControl: false }).setView([43.062, 141.354], 8);

                window.L.tileLayer('https://mt1.google.com/vt/lyrs=r&x={x}&y={y}&z={z}&hl=zh-TW').addTo(map);

                markersLayer = window.L.layerGroup().addTo(map);

                updateMapMarkers();

            };


            const updateMapMarkers = () => {

                if (!map || !markersLayer) return;

                markersLayer.clearLayers();

                const bounds = [];

                currentDayData.value.events.forEach(e => {

                    if (e.lat && e.lng) {

                        window.L.marker([e.lat, e.lng]).addTo(markersLayer);

                        bounds.push([e.lat, e.lng]);

                    }

                });

                if (bounds.length > 0) map.fitBounds(bounds, { padding: [40, 40] });

            };


            const dayExpenses = computed(() => expenses.value.filter(e => e.dayNum === currentDayData.value.dayNum));

            const dayTotalTWD = computed(() => dayExpenses.value.reduce((s, e) => s + (parseInt(e.amountTWD) || 0), 0));


            const handleJpyInput = () => {

                expenseForm.value.amountTWD = Math.round(expenseForm.value.amountJPY * 0.205);

            };


            const saveExpense = async () => {

                const data = { ...expenseForm.value, id: Date.now().toString(), dayNum: currentDayData.value.dayNum };

                if (isCloudSync.value) {

                    await setDoc(doc(fDb, 'artifacts', cloudAppId, 'public', 'data', 'expenses', data.id), data);

                } else {

                    expenses.value.push(data);

                }

                showExpenseModal.value = false;

            };


            const getLuggageByCategory = (c) => luggageList.value.filter(i => i.person === currentLuggagePerson.value && i.category === c);

            const currentPersonPackedCount = computed(() => luggageList.value.filter(i => i.person === currentLuggagePerson.value && i.checked).length);

            const currentPersonTotalCount = computed(() => luggageList.value.filter(i => i.person === currentLuggagePerson.value).length);


            const saveLuggage = async () => {

                const data = { ...luggageForm.value, id: Date.now().toString(), person: currentLuggagePerson.value };

                if (isCloudSync.value) {

                    await setDoc(doc(fDb, 'artifacts', cloudAppId, 'public', 'data', 'luggage', data.id), data);

                } else {

                    luggageList.value.push(data);

                }

                showLuggageModal.value = false;

            };


            const toggleLuggageChecked = async (item) => {

                if (isCloudSync.value) {

                    await setDoc(doc(fDb, 'artifacts', cloudAppId, 'public', 'data', 'luggage', item.id), { ...item });

                }

            };


            onMounted(() => {

                if (MY_FIREBASE_CONFIG) {

                    const app = initializeApp(MY_FIREBASE_CONFIG);

                    fAuth = getAuth(app);

                    fDb = getFirestore(app);

                    signInAnonymously(fAuth).then(() => {

                        isCloudSync.value = true;

                        onSnapshot(collection(fDb, 'artifacts', cloudAppId, 'public', 'data', 'expenses'), s => {

                            expenses.value = s.docs.map(d => d.data());

                        });

                        onSnapshot(collection(fDb, 'artifacts', cloudAppId, 'public', 'data', 'luggage'), s => {

                            luggageList.value = s.docs.map(d => d.data());

                        });

                    });

                } else {

                    const localExp = localStorage.getItem('exp');

                    if (localExp) expenses.value = JSON.parse(localExp);

                    const localLug = localStorage.getItem('lug');

                    if (localLug) luggageList.value = JSON.parse(localLug);

                }

            });


            watch(expenses, (v) => { if(!isCloudSync.value) localStorage.setItem('exp', JSON.stringify(v)); }, { deep: true });

            watch(luggageList, (v) => { if(!isCloudSync.value) localStorage.setItem('lug', JSON.stringify(v)); }, { deep: true });

            watch(currentTab, v => { if(v === 'map') nextTick(initMap); });


            return { 

                currentTab, currentDay, itinerary, currentDayData, selectDay, isCloudSync,

                expenses, dayExpenses, dayTotalTWD, showExpenseModal, expenseForm, handleJpyInput, saveExpense,

                luggagePeople, luggageCategories, currentLuggagePerson, luggageList, getLuggageByCategory,

                showLuggageModal, luggageForm, saveLuggage, toggleLuggageChecked, currentPersonPackedCount, currentPersonTotalCount,

                openExpenseForm: () => showExpenseModal.value = true,

                openLuggageForm: () => { luggageForm.value = { person: currentLuggagePerson.value, category: '行李&證件', name: '', quantity: 1, checked: false }; showLuggageModal.value = true; }

            };

        }

    }).mount('#app');

</script>
